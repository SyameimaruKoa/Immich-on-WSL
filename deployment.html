Step 3: docker-compose.yml の作成

WSL (immich_user) 
cat <<EOF > docker-compose.yml
name: immich
services:
  immich-server:
    container_name: immich_server
    image: ghcr.io/immich-app/immich-server:\${IMMICH_VERSION:-release}
    volumes:
      - \${UPLOAD_LOCATION}:/usr/src/app/upload
      - /etc/localtime:/etc/localtime:ro
    env_file: .env
    environment:
      - NVIDIA_VISIBLE_DEVICES=all
      - NVIDIA_DRIVER_CAPABILITIES=compute,video,utility
    ports: ['2283:2283']
    depends_on: [redis, database]
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: 1
              capabilities: [gpu, video, utility]
    restart: always
  immich-machine-learning:
    container_name: immich_machine_learning
    image: ghcr.io/immich-app/immich-machine-learning:\${IMMICH_VERSION:-release}-cuda
    volumes: [model-cache:/cache]
    env_file: .env
    deploy:
      resources:
        reservations:
          devices: [{driver: nvidia, count: 1, capabilities: [gpu, video, utility]}]
    restart: always
  redis:
    container_name: immich_redis
    image: docker.io/valkey/valkey:8
    restart: always
  database:
    container_name: immich_postgres
    image: ghcr.io/immich-app/postgres:14-vectorchord0.4.3-pgvectors0.2.0
    environment:
      POSTGRES_PASSWORD: \${DB_PASSWORD}
      POSTGRES_USER: \${DB_USERNAME}
      POSTGRES_DB: \${DB_DATABASE_NAME}
      POSTGRES_INITDB_ARGS: '--data-checksums'
    volumes:
      - \${DB_DATA_LOCATION}:/var/lib/postgresql/data
    restart: always
  watchtower:
    container_name: immich_watchtower
    image: containrrr/watchtower
    volumes: [/var/run/docker.sock:/var/run/docker.sock]
    environment:
      - WATCHTOWER_POLL_INTERVAL=86400
      - WATCHTOWER_CLEANUP=true
      - WATCHTOWER_INCLUDE_RESTARTING=true
    restart: always
networks:
  default:
    driver: bridge
    driver_opts:
      com.docker.network.driver.mtu: 1200
volumes:
  model-cache:
EOF

Step 5: Windows側 自動起動設定 (Managed Keep-Alive)

Windows (Notepad等)

Start-Immich.bat という名前で保存し、タスクスケジューラには start 引数付きで登録する。
ポイント: 顔認識やトランスコードの安定性重視で、 WSL内に systemd 管理の keep-alive ユニットを作成して維持する。 

@echo off
setlocal
REM [HELP] ヘルプはこのファイルの下部 (:show_help) にあります

set "DISTRO=Immich-Server"
set "WSL_USER=immich_user"
set "IMMICH_DIR=/home/immich_user/immich"

if "%~1"=="" goto :help_pause
if /I "%~1"=="-h" goto :show_help
if /I "%~1"=="--help" goto :show_help

if /I "%~1"=="start" goto :start
if /I "%~1"=="status" goto :status
if /I "%~1"=="stop-keepalive" goto :stop_keepalive

echo [ERROR] Unknown argument: %~1
echo.
goto :show_help

:start
echo [1/3] Starting Docker in %DISTRO%...
wsl -d "%DISTRO%" -u root -- bash -lc "systemctl start docker 2>/dev/null || service docker start"
if errorlevel 1 exit /b 1

echo [2/3] Starting Immich stack...
wsl -d "%DISTRO%" -u "%WSL_USER%" -- bash -lc "cd '%IMMICH_DIR%' && docker compose up -d"
if errorlevel 1 exit /b 1

echo [3/3] Creating managed keep-alive unit...
wsl -d "%DISTRO%" -u root -- bash -lc "systemd-run --unit=immich-wsl-keepalive --property=Restart=always --property=RestartSec=5 /usr/bin/sleep infinity >/dev/null 2>&1 || true"
exit /b 0

:status
wsl -d "%DISTRO%" -u "%WSL_USER%" -- bash -lc "cd '%IMMICH_DIR%' && docker compose ps"
wsl -d "%DISTRO%" -u root -- bash -lc "printf 'keepalive: '; systemctl is-active immich-wsl-keepalive 2>/dev/null || echo inactive"
exit /b 0

:stop_keepalive
wsl -d "%DISTRO%" -u root -- bash -lc "systemctl stop immich-wsl-keepalive 2>/dev/null || true"
wsl -d "%DISTRO%" -u root -- bash -lc "systemctl reset-failed immich-wsl-keepalive 2>/dev/null || true"
echo keepalive stopped.
exit /b 0

:help_pause
call :show_help
echo.
pause
exit /b 1

:show_help
echo Usage:
echo   %~nx0 start          ^(Docker起動 + Immich起動 + keepalive有効化^)
echo   %~nx0 status         ^(Immich状態 + keepalive状態表示^)
echo   %~nx0 stop-keepalive ^(keepaliveのみ停止^)
echo   %~nx0 -h ^| --help ^(ヘルプ表示^)
echo.
echo Task Scheduler:
echo   Program/script: C:\path\to\Start-Immich.bat
echo   Add arguments: start
exit /b 0

タスクスケジューラ登録 (GUI):

1) タスク スケジューラ → タスクの作成
2) [全般]
  - 名前: Immich Startup
  - 「ユーザーがログオンしているかどうかにかかわらず実行する」
  - 「最上位の特権で実行する」を有効化
3) [トリガー]
  - 「スタートアップ時」または「ログオン時」を追加
4) [操作]
  - 操作: プログラムの開始
  - プログラム/スクリプト: C:\path\to\Start-Immich.bat
  - 引数の追加: start
5) [条件]
  - 必要なら「AC電源時のみ」の制約を解除
6) [設定]
  - 「要求時にタスクを実行する」を有効化

タスクスケジューラ登録 (CLI):

Windows PowerShell / CMD (管理者) 
schtasks /Create /TN "Immich Startup" /SC ONSTART /RL HIGHEST /TR "\"C:\path\to\Start-Immich.bat\" start" /F
schtasks /Run /TN "Immich Startup"
